  <script>

    // ==================== 🎯 在这里修改你预设的文字内容 ====================

    const DEFAULT_TEXTS = `你好，朋友

欢迎来到我的个人小站

这是第二段文字

点击鼠标会依次切换段落

你可以在这里放上你的自我介绍

或者一句喜欢的诗

最后一段，谢谢观看！`;

    // ====================================================================


    // ==================== 🎵 预设音乐文件（放在仓库同目录下）====================

    // 方法1：把音乐文件（比如 music.mp3）放在和 index.html 同一文件夹

    // 然后把下面的 'music.mp3' 改成你的文件名

    const DEFAULT_MUSIC_URL = '';  // 例如: 'background-music.mp3' 或留空不启用

    

    // 如果要使用远程音乐链接，可以填 https://... 开头的网址

    // const DEFAULT_MUSIC_URL = 'https://example.com/music.mp3';

    // ====================================================================


    const stage = document.getElementById("stage");

    const displayText = document.getElementById("displayText");

    const textInput = document.getElementById("textInput");

    const fontFamily = document.getElementById("fontFamily");

    const textColor = document.getElementById("textColor");

    const fontSize = document.getElementById("fontSize");

    const textAlign = document.getElementById("textAlign");

    const musicFile = document.getElementById("musicFile");

    const bgm = document.getElementById("bgm");

    const toggleMusic = document.getElementById("toggleMusic");

    const resetTextBtn = document.getElementById("resetText");

    

    let paragraphs = [];

    let currentIndex = -1;

    let musicReady = false;

    

    // 检测URL参数，判断是否进入编辑模式

    function checkEditMode() {

      const urlParams = new URLSearchParams(window.location.search);

      const editParam = urlParams.get('edit');

      if (editParam === '1') {

        document.body.classList.add('edit-mode');

      } else {

        document.body.classList.remove('edit-mode');

      }

    }

    

    function parseParagraphs() {

      paragraphs = textInput.value

        .split("\n")

        .map((t) => t.trim())

        .filter(Boolean);

      if (currentIndex >= paragraphs.length) {

        currentIndex = paragraphs.length - 1;

      }

    }

    

    function updateStyle() {

      document.documentElement.style.setProperty("--font-family", fontFamily.value);

      document.documentElement.style.setProperty("--text-color", textColor.value);

      document.documentElement.style.setProperty("--font-size", `${Number(fontSize.value) || 36}px`);

      document.documentElement.style.setProperty("--text-align", textAlign.value);

    }

    

    function showNextText() {

      if (!paragraphs.length) {

        displayText.textContent = "";

        displayText.classList.remove("show");

        return;

      }

      if (currentIndex < paragraphs.length - 1) {

        currentIndex += 1;

      }

      displayText.classList.remove("show");

      requestAnimationFrame(() => {

        displayText.textContent = paragraphs[currentIndex];

        displayText.classList.add("show");

      });

    }

    

    function resetText() {

      parseParagraphs();

      currentIndex = -1;

      displayText.textContent = "";

      displayText.classList.remove("show");

    }

    

    // 预设音乐加载（如果配置了默认音乐）

    function loadDefaultMusic() {

      if (DEFAULT_MUSIC_URL && DEFAULT_MUSIC_URL.trim() !== '') {

        bgm.src = DEFAULT_MUSIC_URL;

        musicReady = true;

        toggleMusic.textContent = "播放音乐";

        // 可选：自动播放（注意浏览器策略，可能需要用户交互）

        // 为了更好的体验，不自动播放，让用户点击播放按钮

      }

    }

    

    // 保存设置到 localStorage（非编辑模式下也保留用户的选择）

    function saveSettings() {

      const settings = {

        fontFamily: fontFamily.value,

        textColor: textColor.value,

        fontSize: fontSize.value,

        textAlign: textAlign.value

      };

      localStorage.setItem('pageSettings', JSON.stringify(settings));

    }

    

    function loadSettings() {

      const saved = localStorage.getItem('pageSettings');

      if (saved) {

        try {

          const settings = JSON.parse(saved);

          if (settings.fontFamily) fontFamily.value = settings.fontFamily;

          if (settings.textColor) textColor.value = settings.textColor;

          if (settings.fontSize) fontSize.value = settings.fontSize;

          if (settings.textAlign) textAlign.value = settings.textAlign;

          updateStyle();

        } catch(e) {}

      }

    }

    

    // 保存文字内容到 localStorage（非编辑模式下）

    function saveTextContent() {

      localStorage.setItem('savedTexts', textInput.value);

    }

    

    function loadTextContent() {

      const saved = localStorage.getItem('savedTexts');

      if (saved && saved.trim() !== '') {

        textInput.value = saved;

      } else {

        textInput.value = DEFAULT_TEXTS;

      }

      parseParagraphs();

    }

    

    // 保存音乐URL

    function saveMusicURL(url) {

      if (url) localStorage.setItem('musicURL', url);

    }

    

    function loadMusicURL() {

      const saved = localStorage.getItem('musicURL');

      if (saved && saved.trim() !== '') {

        bgm.src = saved;

        musicReady = true;

        toggleMusic.textContent = "播放音乐";

      } else if (DEFAULT_MUSIC_URL && DEFAULT_MUSIC_URL.trim() !== '') {

        loadDefaultMusic();

      }

    }

    

    // 事件绑定

    stage.addEventListener("click", showNextText);

    

    [fontFamily, textColor, fontSize, textAlign].forEach((el) => {

      el.addEventListener("input", () => {

        updateStyle();

        saveSettings();

      });

      el.addEventListener("change", () => {

        updateStyle();

        saveSettings();

      });

    });

    

    textInput.addEventListener("input", () => {

      parseParagraphs();

      saveTextContent();

    });

    

    resetTextBtn.addEventListener("click", () => {

      resetText();

      saveTextContent();

    });

    

    musicFile.addEventListener("change", () => {

      const file = musicFile.files && musicFile.files[0];

      if (!file) return;

      const fileURL = URL.createObjectURL(file);

      bgm.src = fileURL;

      musicReady = true;

      toggleMusic.textContent = "播放音乐";

      saveMusicURL(fileURL);

    });

    

    toggleMusic.addEventListener("click", async () => {

      if (!musicReady) {

        // 如果没有音乐，提示用户

        alert("请先通过「添加音乐」上传音乐文件");

        return;

      }

      if (bgm.paused) {

        try {

          await bgm.play();

          toggleMusic.textContent = "暂停音乐";

        } catch (err) {

          console.error("音乐播放失败:", err);

          alert("播放失败，请点击页面任意位置后重试");

        }

      } else {

        bgm.pause();

        toggleMusic.textContent = "播放音乐";

      }

    });

    

    // 初始化

    checkEditMode();

    loadSettings();

    loadTextContent();

    loadMusicURL();

    updateStyle();

    

    // 如果当前不是编辑模式，给访客一个友好的提示（显示几秒后淡出）

    if (!document.body.classList.contains('edit-mode')) {

      const tip = document.createElement('div');

      tip.textContent = '✨ 点击屏幕任意位置，切换文字 ✨';

      tip.style.position = 'fixed';

      tip.style.bottom = '20px';

      tip.style.left = '50%';

      tip.style.transform = 'translateX(-50%)';

      tip.style.color = 'rgba(255,255,255,0.6)';

      tip.style.fontSize = '14px';

      tip.style.backgroundColor = 'rgba(0,0,0,0.5)';

      tip.style.padding = '6px 12px';

      tip.style.borderRadius = '20px';

      tip.style.zIndex = '999';

      tip.style.pointerEvents = 'none';

      tip.style.whiteSpace = 'nowrap';

      document.body.appendChild(tip);

      setTimeout(() => {

        tip.style.opacity = '0';

        tip.style.transition = 'opacity 1s';

        setTimeout(() => tip.remove(), 1000);

      }, 3000);

    }

  </script>
